<script>
let bank=[
{q:"한국의 수도는?",o:["서울","부산","대구","인천"],a:0},
{q:"물은?",o:["water","fire","air","food"],a:0},
{q:"밥을 먹다?",o:["sleep","eat","run","write"],a:1}
];

bank.sort(()=>Math.random()-0.5);

let quiz=document.getElementById("quiz");

bank.forEach((x,i)=>{
let html=`<div class='q'><p>${x.q}</p>`;
x.o.forEach((op,j)=>{
html+=`<label><input type='radio' name='q${i}' value='${j}'> ${op}</label><br>`;
});
html+="</div>";
quiz.innerHTML+=html;
});

function submitTest(){
let score=0;
bank.forEach((x,i)=>{
let sel=document.querySelector(`input[name=q${i}]:checked`);
if(sel && sel.value==x.a) score++;
});
document.getElementById("t").innerText="Score: "+score;
}
</script>
